7. List passengers that are waitlisted including the name of the train.

SELECT first_name, last_name, train_name FROM Passenger P, Booked B, Train T WHERE B.status = 'WaitL' AND B.passenger_ssn = P.ssn AND B.train_number = T.train_number;



8. List passengers that have '605' phone area code in descending order.

SELECT first_name, last_name FROM Passenger P WHERE P.phone LIKE '%605%' ORDER BY P.first_name DESC, P.last_name DESC;



9. List name of passengers that are traveling on Thursdays in ascending order.

SELECT first_name, last_name FROM Passenger P, Booked B, Train T WHERE T.available_on LIKE '%Thursday%' AND B.passenger_ssn = P.ssn AND T.train_number = B.train_number ORDER BY first_name ASC, last_name ASC;
